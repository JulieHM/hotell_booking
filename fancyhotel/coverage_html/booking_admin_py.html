58 def overview(self, request, NUMBER_OF_DAYS): 

59 """ 

60 A view, rendering an HTML-table with rooms as rows and days as columns 

61 

62  

63 Example of busy_dates, using NUMBER_OF_DAYS = 7: 

64 busy_dates == { 

65 101 : [None, None, None, ['Berg, Andreas', 10], ['Berg, Andreas', 10], None, None] 

66 102 : [None, None, ['Nordmann, Ola', 12], None, ['Nordmann, Kari', 13], None, None] 

67 } 

68 """ 

69 

70 busy_dates = {} 

71 header_row_dates = [] 

72 today = datetime.date.today() 

73 end_of_period = today + datetime.timedelta(NUMBER_OF_DAYS - 1) 

74 

75 rooms = Hotelroom.objects.all() 

76 

77 # Create header-row 

78 for i in range(NUMBER_OF_DAYS): 

79 date = today + datetime.timedelta(i) 

80 header_row_dates.append(str(date.day) + '/' + str(date.month)) 

81 

82 # Fill busy_dates with NUMBER_OF_DAYS bolean values 

83 for room in rooms: 

84 busy_dates[room.roomNumber] = [None] * NUMBER_OF_DAYS 

85 

86 # Find all bookings in time period: 

87 bookings = Booking.objects.filter(dateEnd__gte=today).filter( 

88 Q(dateEnd__lte=end_of_period) | Q(dateStart__lt=end_of_period)) 

89 

90 # Go through bookings,  

91 for booking in bookings: 

92 start_index = (booking.dateStart - today).days 

93 end_index = (booking.dateEnd - today).days 

94 

95 if start_index < 0: 

96 start_index = 0 

97 if end_index > NUMBER_OF_DAYS: 

98 end_index = NUMBER_OF_DAYS 

99 

100 # Adds a booking to busy_dates. The overview shows each booking starting the day the customer 

101 # arrives, and then have length = as many nights as the customer stays. This makes sense from 

102 # a business perspective, as the hotel can have another customer checking in while the first 

103 # customer checks out 

104 for i in range(start_index, end_index): 

105 busy_dates[booking.room.roomNumber][i] = dict( 

106 name = (booking.lastName + ', ' + booking.firstName), 

107 id = booking.id) 

108 

109 context = dict( 

110 self.admin_site.each_context(request), # Common variables for rendering the admin template 

111 # Fill in values here 

112 header_row_dates = header_row_dates, 

113 busy_dates = busy_dates, 

114 weekdays_until_saturday = 6 - today.weekday(), 

115 weekdays_until_sunday = 7 - today.weekday() 

116 ) 

117 

118 return TemplateResponse(request, 'admin/overview.html', context)
